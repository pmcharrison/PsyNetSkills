evidence-workflow-demo/README.md

                md
                513 bytes
                
              
              
                
                  

                    
This static browser fixture demonstrates the preferred evidence workflow for participant-facing review artifacts.


Run the Playwright flow from this directory:


npm install
npm test

The test asserts participant behavior and writes screenshots to ../../evidence/screenshots/.
For recording, run the same test headed at a readable pace while ffmpeg
captures the browser display:


PARTICIPANT_FLOW_SCREENSHOTS=0 PARTICIPANT_FLOW_STEP_MS=450 npm run test:headed

<body>
  <main>
    <section id="instructions" data-page="instructions">
      <p class="muted">Participant flow</p>
      <h1>Melody matching micro-study</h1>
      <p>You will complete two short trials. Listen to each symbolic melody, choose its contour, and review the feedback.</p>
      <label><input id="consent" type="checkbox"> I understand this demonstration task.</label>
      <div><button id="start" class="primary" disabled>Start task</button></div>
    </section>

    <section id="trial" class="hidden" data-page="trial">
      <p id="progress" class="muted"></p>
      <h1 id="trial-title"></h1>
      <div id="stimulus" class="stimulus" aria-live="polite"></div>
      <button id="play" class="primary">Play melody</button>
      <div class="card" aria-label="Choices">
        <button class="choice" data-answer="rising" disabled>Rising</button>
        <button class="choice" data-answer="falling" disabled>Falling</button>
        <button class="choice" data-answer="flat" disabled>Flat</button>
      </div>
      <p id="feedback" class="feedback" aria-live="polite"></p>
      <button id="next" class="primary" disabled>Next</button>
    </section>

    <section id="complete" class="hidden" data-page="complete">
      <p class="muted">Complete</p>
      <h1>Thank you</h1>
      <p id="score"></p>
      <p>Your responses were saved in the browser state for review.</p>
      <pre><code id="saved-data"></code></pre>
    </section>
  </main>
  <script>
    const trials = [
      { title: 'Trial 1: rising contour', notes: 'C D E', answer: 'rising' },
      { title: 'Trial 2: falling contour', notes: 'G F E', answer: 'falling' },
    ];
    const state = { index: 0, responses: [] };
    const $ = (id) => document.getElementById(id);
    const show = (id) => {
      for (const section of document.querySelectorAll('main > section')) section.classList.add('hidden');
      $(id).classList.remove('hidden');
    };
    const renderTrial = () => {
      const trial = trials[state.index];
      $('progress').textContent = `Trial ${state.index + 1} of ${trials.length}`;
      $('trial-title').textContent = trial.title;
      $('stimulus').textContent = trial.notes;
      $('feedback').textContent = '';
      $('next').disabled = true;
      document.querySelectorAll('.choice').forEach((button) => { button.disabled = true; button.dataset.selected = 'false'; });
      show('trial');
    };
    $('consent').addEventListener('change', (event) => { $('start').disabled = !event.target.checked; });
    $('start').addEventListener('click', renderTrial);
    $('play').addEventListener('click', () => {
      $('play').disabled = true;
      $('feedback').textContent = 'Playing...';
      setTimeout(() => {
        $('feedback').textContent = 'Choose the contour now.';
        document.querySelectorAll('.choice').forEach((button) => { button.disabled = false; });
        $('play').disabled = false;
      }, 450);
    });
    document.querySelectorAll('.choice').forEach((button) => {
      button.addEventListener('click', () => {
        const trial = trials[state.index];
        const correct = button.dataset.answer === trial.answer;
        document.querySelectorAll('.choice').forEach((choice) => { choice.disabled = true; choice.dataset.selected = String(choice === button); });
        state.responses.push({ trial: state.index + 1, answer: button.dataset.answer, correct });
        $('feedback').textContent = correct ? 'Correct - the contour matches.' : `Not quite - this one was ${trial.answer}.`;
        $('next').disabled = false;
      });
    });
    $('next').addEventListener('click', () => {
      state.index += 1;
      if (state.index < trials.length) {
        renderTrial();
      } else {
        const correct = state.responses.filter((response) => response.correct).length;
        $('score').textContent = `Score: ${correct} / ${trials.length}`;
        $('saved-data').textContent = JSON.stringify(state.responses, null, 2);
        window.participantResponses = state.responses;
        show('complete');
      }
    });
  </script>
</body>
</html>
